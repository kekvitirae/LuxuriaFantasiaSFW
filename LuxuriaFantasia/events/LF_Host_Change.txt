# Bad Genetics fix for create_character births

character_event = {
	id = LFHostChange.3
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_game_rule = { name = lf_challenges value = lf_challenges_genes }
		OR = {
			mother_even_if_dead = {
				any_owned_bloodline = {
					has_bloodline_flag = lf_bad_genetics
				}
			}
			father_even_if_dead = {
				any_owned_bloodline = {
					has_bloodline_flag = lf_bad_genetics
				}
			}
			true_father_even_if_dead = {
				any_owned_bloodline = {
					has_bloodline_flag = lf_bad_genetics
				}
			}
		}		
		age == 0
	}
	immediate = {
		random_list = {
			75 = {
				remove_trait = fair
				remove_trait = genius
				remove_trait = quick
				remove_trait = strong
				add_trait = lf_bad_genetics
			}
			25 = {
			}
		}
	}
}
