# Gain the mage trait
narrative_event = {
	id = LFMagic.3
	picture = lf_mage_trait
	title = LFMagic3T
	is_triggered_only = yes
	desc = {
		trigger = {
			is_kitsune = yes
		}
		text = LFMagic3Kitsune
	}
	desc = {
		trigger = {
			is_fairy = yes
		}
		text = LFMagic3Fairy
	}
	desc = {
		trigger = {
			OR = {
				is_elf = yes
				trait = houseirae_drow
				trait = houseirae_halfelf
			}
		}
		text = LFMagic3Elf
	}
	desc = {
		trigger = {
			NOR = {
				trait = houseirae_drow
				trait = houseirae_halfelf
			}
			is_not_kitsune = yes
			is_not_elf = yes
			is_not_fairy = yes
		}
		text = LFMagic3
	}

	option = { # Accept the trait
		name = LFMagic3TTA
		lf_add_mage = yes
		if = {
			limit = {
				NOR = {
					trait = houseirae_drow
					trait = houseirae_halfelf
				}
				is_not_kitsune = yes
				is_not_elf = yes
				is_not_fairy = yes
			}
			health = -1
		}
		custom_tooltip = {
			text = increased_mana_tt
			set_max_mana = yes
		}
	}
	
	option = { # Come back later
		name = LFMagic3TTB
	}
}
