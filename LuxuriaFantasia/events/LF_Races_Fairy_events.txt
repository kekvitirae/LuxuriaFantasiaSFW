	picture = fairies_dancing
	title = lf_fairies2
	is_triggered_only = yes

		trigger = { 
			is_female = yes
		}
		text = LF_FairyFemale1
	}
	desc = { # Male specific
		trigger = {
			is_female = no
		}
		text = LF_FairyMale1
	}
	option = { # King/Queen of the Fae
		name = {
			trigger = {
				is_female = no
			}
			text = LF_FairyMale2
		}
		name = {
			trigger = {
				is_female = yes
			}
			text = LF_FairyFemale2
		}
		set_character_flag = lf_fairy
		lf_change_race = yes
	}
	option = { # Everyone is fae
		name = LF_Fairy2TTC
		lf_change_race = yes
		custom_tooltip = {
			text = LFAllRealmCharacters
		}
		hidden_tooltip = {
			set_character_flag = lf_fairy
			lf_change_race = yes
			any_realm_character = {
				limit = {
					any_liege = { character = ROOT }
				}
				set_character_flag = lf_fairy
				lf_change_race = yes
			}
		}
	}	
	after = {
		if = {
			limit = {
				has_global_flag = LFGenderBendOn
			}
			set_character_flag = lf_fairy_genderbend
		}
		character_event = { id = LFMultiplayer.1000 }
	}
}

narrative_event = { # You spy a girl drowning
	id = LF_Fairy.3
	ai = no
	title = LF_Fairy3T
	picture = lf_bonny_swans_drown
	desc = LF_Fairy3
	is_triggered_only = yes
	trigger = {
		trait = poet
		is_fairy = yes
		lf_has_fairies = yes
		NOT = { has_global_flag = bonny_swan_happened }
	}
	
	immediate = {
		set_global_flag = bonny_swan_happened
		create_character = {
			random_traits = no
			female = yes
			name = Anne
			age = 18
			religion = ROOT
			culture = ROOT
			health = 8.0
			historical = yes
			trait = flamboyant_schemer
			trait = fair
			trait = cruel
			trait = deceitful
			flag = special_npc		
			attributes = {
				diplomacy = 3 
				intrigue = 6 
				martial = 3
				stewardship = 3
				learning = 3 
			}
		}
		new_character = {
			save_event_target_as = bonny_swan_murderer
		}
		create_character = {
			random_traits = no
			female = yes
			age = 16
			religion = ROOT
			culture = ROOT
			health = 8.0
			trait = charismatic_negotiator
			trait = fair
			trait = chaste
			trait = kind
			flag = special_npc		
			attributes = {
				diplomacy = 6 
				intrigue = 3 
				martial = 3
				stewardship = 3
				learning = 3 
			}
		}
		new_character = {
			save_event_target_as = bonny_swan
		}		
	}
	
	option = { # I must help the girl!
		name = LF_Fairy3TTA
		hidden_tooltip = {
			character_event = { id = LF_Fairy.31 } # Hides the portraits
		}
		narrative_event = { id = LF_Fairy.4 }
	}
	
	option = { # How tragic.
		name = LF_Fairy3TTB
		hidden_tooltip = {
			character_event = { id = LF_Fairy.31 } # Hides the portraits
		}
	}
}

	id = LF_Fairy.31

	}
}
