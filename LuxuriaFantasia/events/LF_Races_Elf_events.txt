	id = LF_Elf.10
	title = LF_ElfT	
	is_triggered_only = yes
	picture = {
		trigger = {
			is_female = yes
		}
		picture = lf_elf_start_female
	}
	picture = {
		trigger = {
			is_female = no
		}
		picture = lf_elf_start_male
	}
	desc = LF_Elf10
	option = {
		name = LF_Elf10A
		set_character_flag = lf_elf
		lf_change_race = yes
	}
	option = {
		name = LF_Elf10B
		lf_change_race = yes
		custom_tooltip = {
			text = LFAllRealmCharacters
		}
		hidden_tooltip = {
			set_character_flag = lf_elf
			lf_change_race = yes
			any_realm_character = {
				limit = {
					any_liege = { character = ROOT }
				}
				set_character_flag = lf_elf
				lf_change_race = yes
			}
		}
	}	
	after = {
		if = {
			limit = {
				has_global_flag = LFGenderBendOn
			}
			set_character_flag = lf_elf_genderbend
		}
		character_event = { id = LFMultiplayer.1000 }
	}
}

narrative_event = { # Unicorn
	id = LF_Elf.20
	ai = no
	picture = GFX_unicorn
	is_triggered_only = yes
	title = LF_Elf2T
	desc = LF_Elf2
	trigger = {
		lf_has_elves = yes
		is_elf = yes
		OR = {
			trait = chaste
			trait = just
			trait = kind
		}
		practical_age < 45
		is_pregnant = no
	}

	immediate = {
		create_character = {
			random_traits = no
			female = no
			name = Wynn
			age = 16
			religion = ROOT
			culture = horse
			race = horse
			health = 8.0
			historical = yes
			trait = flamboyant_schemer
			trait = fair
			trait = strong
			trait = kind
			trait = just
			trait = lf_unicorn
			flag = special_npc		
			attributes = {
				diplomacy = 6 
				intrigue = 6 
				martial = 6
				stewardship = 6
				learning = 6 
			}
		}
		new_character = {
			set_character_flag = lf_unicorn
			save_event_target_as = unicorn
		}
	}
	option = { # He is so majestic. I must bring him to court!
		name = LF_Elf2TTA
		event_target:unicorn = { }
	}
	option = { # He is so... big.
		name = LF_Elf2TTB
		trigger = {
			has_bestiality_content = yes
			OR = {
				AND = {
					is_female = no
					trait = homosexual
				}
				is_female = yes
			}
		}
		event_target:unicorn = { }
		add_lover = event_target:unicorn
		hidden_tooltip = {
			if = {
				limit = {
					is_female = yes
					is_married = no
					is_pregnant = no
				}
				impregnate = event_target:unicorn
			}
			else_if = {
				limit = {
					is_female = yes
					is_married = yes
					is_pregnant = no
				}
				impregnate_cuckoo = event_target:unicorn
			}
		}
	}
	option = {
		name = LF_Elf2TTC
		trigger = {
			OR = {
				trait = cruel
				trait = impaler
				trait = is_fat
				trait = is_malnourished
				trait = gluttonous
			}
		}
		event_target:unicorn = {
			death = { death_reason = death_execution_eaten killer = ROOT }
		}
		add_character_modifier = {
			name = unicorn_meat
			duration = -1
		}
	}
		
}

# .32 Elven Immortality for spawned characters

	id = LF_Elf.32
	min_age = 16
	hide_window = yes
	is_triggered_only = yes # on_yearly_pulse
	trigger = {
		lf_has_elves = yes
		is_elf = yes
		NOT = { trait = lf_elf_immortality }
		NOT = { has_game_rule = { name = lf_challenges value = lf_challenges_random } }
	}
	immediate = {
		add_trait = lf_elf_immortality
	}
}
