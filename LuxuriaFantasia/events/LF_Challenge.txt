character_event = {
	id = LFChallenge.1
	desc = {
		trigger = {
			has_character_flag = lf_randomize
		}
		text = LFChallenge1Random
	}
	desc = {
		trigger = {
			has_character_flag = lf_stay
		}
		text = LFChallenge1Stay
	}
	picture = lf_magic_user_female
	is_triggered_only = yes
	trigger = {
		has_game_rule = { name = lf_challenges value = lf_challenges_random }
		ai = no
	}
	immediate = {
		random_list = {
			50 = {
				set_character_flag = lf_randomize
			}
			50 = {
				set_character_flag = lf_stay
			}
		}
	}
	option = {
		name = {
			trigger = {
				has_character_flag = lf_stay
			}
			text = LFChallenge1TTA
		}
		name = {
			trigger = {
				has_character_flag = lf_random
			}
			text = LFChallenge1TTB
		}
		
		hidden_tooltip = {
			if = {
				limit = {
					has_character_flag = lf_randomize
				}
				random_playable_ruler = {
					limit = {
						ai = no
						is_adult = yes
					}
					save_event_target_as = random_ruler
					if = {
						limit = {
							is_married = yes
						}
						spouse = {
							save_event_target_as = random_spouse
						}
					}
					create_character = {
						dynasty = THIS
						religion = THIS
						culture = THIS
						race = THIS
						age = 0
					}
					new_character = {
						if = {
							limit = {
								event_target:random_ruler = {
									is_female = yes
								}
							}
							set_mother = event_target:random_ruler
						}
						else = {
							set_father = event_target:random_ruler
						}
						if = {
							limit = {
								event_target:random_ruler = {
									is_female = yes
									is_married = yes
								}
							}
							set_father = event_target:random_spouse
						}
						else_if = {
							limit = {
								event_target:random_ruler = {
									is_female = no
									is_married = yes
								}
							}
							set_mother = event_target:random_spouse
						}
						ROOT = {
							set_player_character = PREV
						}
					}
				}
			}
			sound_effect = child_born
		}
	}
}
