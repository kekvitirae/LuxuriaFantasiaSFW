title_lf_owo = { # Offsets the fertility penalty
	is_high_prio = no
	realm_in_name = yes
	dignity = 0.9
	grant_limit = 1
	opinion_effect = 10
	revoke_allowed = yes
	

	allowed_to_hold = {
		prisoner = no
		is_female = yes
		is_adult = yes
		NOT = { trait = incapable }
		is_neko = yes
		is_futa = yes
	}
	
	allowed_to_grant = {
		is_neko = yes
	}
	revoke_trigger = {
		
	}

	gain_effect = {
		add_character_modifier = {
			name = modifier_lf_owo
			duration = -1
		}
	}

	lose_effect = {
		remove_character_modifier = modifier_lf_owo
	}
	retire_effect = {
		remove_character_modifier = modifier_lf_owo
	}	
	message = yes
}
