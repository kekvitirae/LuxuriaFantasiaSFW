lf_is_not_same_race = {
	OR = {
		AND = {
			NOT = { trait = lf_elf }
			ROOT = {
				trait = lf_elf
			}
		}
		AND = {
			NOT = { trait = lf_orc }
			ROOT = {
				trait = lf_orc
			}
		}
		AND = {
			NOT = { trait = lf_fairy }
			ROOT = {
				trait = lf_fairy
			}
		}
		AND = {
			NOT = { trait = lf_kitsune }
			ROOT = {
				trait = lf_kitsune
			}
		}
		AND = {
			NOT = { trait = lf_neko }
			ROOT = {
				trait = lf_neko
			}
		}
		AND = {
			NOR = { 
				trait = dwse_dragon_adult
				trait = dwse_lizard
				trait = dwse_shapeshifted
				trait = dwse_dragon_child
			}
			ROOT = {
				OR = { 
					trait = dwse_dragon_adult
					trait = dwse_lizard
					trait = dwse_shapeshifted
					trait = dwse_dragon_child
				}
			}
		}
		AND = {
			NOR = { 
				trait = lf_demon
				trait = lf_halfdemon
				trait = lf_fallen_angel
			}
			ROOT = {
				OR = { 
					trait = lf_demon
					trait = lf_halfdemon
					trait = lf_fallen_angel
				}
			}
		}
		AND = {
			NOT = { trait = lf_angel }
			ROOT = {
				trait = lf_angel
			}
		}
	}
}
