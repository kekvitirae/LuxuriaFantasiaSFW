yearly_mana_increase = {
	set_variable = {
		which = mana_increase
		which = max_mana
	}
	if = {
		limit = {
			has_character_flag = lf_lenora_blessed
		}
		divide_variable = {
			which = mana_increase
			value = 8
		}
	}
	else = {
		divide_variable = {
			which = mana_increase
			value = 10
		}
	}
	change_variable = {
		which = current_mana
		which = mana_increase
	}
	
	if = {
		limit = {
			check_variable = {
				which = current_mana
				which >= max_mana
			}
		}
		set_variable = {
			which = current_mana
			which = max_mana
		}
	}
}

set_max_mana = {
	set_variable = {
		which = max_mana
		value = 0
	}
	
	if = {
		limit = {
			is_mage = yes
		}
		change_variable = {
			which = max_mana
			value = 50
		}
	}
	if = {
		limit = {
			OR = {
				trait = houseirae_drow
				is_kitsune = yes
				is_elf = yes
				is_fairy = yes
			}
		}
		change_variable = {
			which = max_mana
			value = 50
		}
	}
	if = {
		limit = {
			trait = houseirae_halfelf
		}
		change_variable = {
			which = max_mana
			value = 25
		}
	}
	if = {
		limit = {
			has_character_flag = lf_lenora_blessed
		}
		change_variable = {
			which = max_mana
			value = 25
		}
	}
		
	if = {
		limit = {
			NOT = {
				has_character_flag = lf_mana_set
			}
		}
		set_variable = {
			which = starting_mana
			which = max_mana
		}
		divide_variable = {
			which = starting_mana
			value = 2
		}
		set_variable = {
			which = current_mana
			which = starting_mana
		}
		set_character_flag = lf_mana_set
	}
	if = {
		limit = {
			NOT = {
				has_character_modifier = lf_mana_stores
			}
		}
		add_character_modifier = {
			name = lf_mana_stores
			duration = -1
		}
	}
}
