lf_has_potions = {
	OR = {
		AND = {
			ai = no
			OR = {
				family_palace = {
					has_building = lf_potion_seller_fp
				}
				capital_holding = {
					OR = {
						has_building = lf_potion_seller_castle # Castle
						has_building = lf_potion_seller_city # City
						has_building = lf_potion_seller_tribal # Tribal
						has_building = lf_potion_seller_nomad # Nomad
					}
				}
			}
		}
		AND = {
			ai = yes
			location = {
				OR = {
					has_building = lf_potion_seller_castle # Castle
					has_building = lf_potion_seller_city # City
					has_building = lf_potion_seller_tribal # Tribal
					has_building = lf_potion_seller_nomad # Nomad
				}
			}
		}
	}
}
