lf_impregnate_check = {
	#
	# Checking if either lovers are sterile
	#

	if = {
		limit = {
			OR = {
				event_target:lf_sex_partner = { fertility <= 0 }
				event_target:lf_sex_player = { fertility <= 0 }
				event_target:lf_sex_player = { same_sex = event_target:lf_sex_partner }
			}
		}
		break = yes
	}
	
	else_if = {
		event_target:lf_sex_player = {
			set_variable = { 
				which = lf_fertility 
				value = 0 
			}
			export_to_variable = { 
				which = lf_fertility 
				value = fertility 
			}
			export_to_variable = { 
				which = lf_fertility_from 
				value = fertility 
				who = event_target:lf_sex_partner 
			}
			multiply_variable = {
				which = lf_fertility
				value = 100
			}
			multiply_variable = {
				which = lf_fertility_from
				value = 100
			}
			change_variable = { 
				which = lf_fertility
				which = lf_fertility_from
			}
			divide_variable = { 
				which = lf_fertility 
				value = 2 
			}
		}
	
		if = {
			limit = {
				event_target:lf_sex_player = {
					OR = {
						AND = {
							is_married = no
							is_consort = no
						}
						is_consort = event_target:lf_sex_partner
						is_married = event_target:lf_sex_partner
					}
				}
			}
			
			# Checking fertility for randomness
			event_target:lf_sex_player = {
				if = {
					limit = {
						check_variable = { which = lf_fertility value >= 0 }
						NOT = { check_variable = { which = lf_fertility value = 25 } }
					}
					event_target:lf_sex_player = {
						random = {
							chance = 25
							if = {
								limit = {
									is_female = yes
								}
								impregnate_cuckoo = event_target:lf_sex_partner
							}
							else = {
								event_target:lf_sex_partner = {
									impregnate_cuckoo = event_target:lf_sex_player
								}
							}
						}
					}
				}
				else_if = {
					limit = {
						check_variable = { which = lf_fertility value >= 25 }
						NOT = { check_variable = { which = lf_fertility value = 50 } }
					}
					event_target:lf_sex_player = {
						random = {
							chance = 50
							if = {
								limit = {
									is_female = yes
								}
								impregnate_cuckoo = event_target:lf_sex_partner
							}
							else = {
								event_target:lf_sex_partner = {
									impregnate_cuckoo = event_target:lf_sex_player
								}
							}
						}
					}
				}
				else_if = {
					limit = {
						check_variable = { which = lf_fertility value >= 50 }
						NOT = { check_variable = { which = lf_fertility value = 75 } }
					}
					event_target:lf_sex_player = {
						random = {
							chance = 75
							if = {
								limit = {
									is_female = yes
								}
								impregnate_cuckoo = event_target:lf_sex_partner
							}
							else = {
								event_target:lf_sex_partner = {
									impregnate_cuckoo = event_target:lf_sex_player
								}
							}
						}
					}
				}
				else_if = {
					limit = {
						check_variable = { which = lf_fertility value >= 75 }
					}
					event_target:lf_sex_player = {
						if = {
							limit = {
								is_female = yes
							}
							impregnate_cuckoo = event_target:lf_sex_partner
						}
						else = {
							event_target:lf_sex_partner = {
								impregnate_cuckoo = event_target:lf_sex_player
							}
						}
					}
				}
			}
		}

		else_if = {
			limit = { 
				event_target:lf_sex_player = {
					OR = {
						AND = {
							is_married = yes
							NOT = { is_married = event_target:lf_sex_partner }
						}
						AND = {
							is_consort = yes
							NOT = { is_consort = event_target:lf_sex_partner }
						}
					}
				}
			}
			# Checking fertility for randomness
			event_target:lf_sex_player = {
				if = {
					limit = {
						check_variable = { which = lf_fertility value >= 0 }
						NOT = { check_variable = { which = lf_fertility value = 25 } }
					}
					random = {
						chance = 25
						if = {
							limit = {
								is_female = yes
							}
							impregnate_cuckoo = event_target:lf_sex_partner
						}
						else = {
							event_target:lf_sex_partner = {
								impregnate_cuckoo = event_target:lf_sex_player
							}
						}
					}
				}
				else_if = {
					limit = {

lf_portrait_change = {
	hidden_tooltip = {
		lf_remove_all_portraits = yes
		#
		# Will prioritize normal racial traits first
		#
		if = {
			limit = {
				is_kitsune = yes
			}
			lf_add_random_kitsune_portrait = yes
		}
		else_if = {
			limit = {
				is_neko = yes
			}
			lf_add_random_neko_portrait = yes
		}
		else_if = {
			limit = {
				is_elf = yes
			}
			lf_add_random_elf_portrait = yes
		}
		else_if = {
			limit = {
				is_orc = yes
			}
			lf_add_random_orc_portrait = yes
		}
		else_if = {
			limit = {
				is_fairy = yes
			}
			lf_add_random_fairy_portrait = yes
		}
		else_if = {
			limit = {
				trait = dwse_lizard
			}
			add_random_dwse_lizard_portrait = yes
		}
		#
		# Will prioritize mixed breed portraits last
		#
		else_if = {
			limit = {
				is_demon = yes
			}
			lf_add_random_demon_portrait = yes
		}
		else_if = {
			limit = {
				is_angel = yes
			}
			lf_add_random_angel_portrait = yes
		}
	}
}

					}

		event_target:lf_sex_partner = {

			set_variable = { which = lf_fertility_from value = 0 }

lf_slave_value = {
	event_target:lf_slave_being_sold = {
	
		# Sets the initial value
		set_variable = { which = lf_slave_sell_price value = 5 }
	
		# Increases or decreases the value based on traits
		if = {
			limit = {
				OR = {
					trait = administrator
					trait = architect
					trait = duelist
					trait = gamer
					trait = gardener
					trait = hedonist
					trait = hunter
					trait = impaler
					trait = schemer
					trait = seducer
					trait = seductress
					trait = mystic
					trait = scholar
					trait = socializer
					trait = strategist
					trait = theologian
				}
			}
			change_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				OR = {
					is_kitsune = yes
					is_neko = yes
					is_demon = yes
					is_angel = yes
					is_elf = yes
					is_orc = yes
					is_fairy = yes
					trait = dwse_lizard
				}
			}
			change_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				is_mage = yes
			}
			change_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				trait = lf_asexual
			}
			subtract_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				OR = {
					trait = temperate
					trait = lustful
					trait = diligent
					trait = patient
					trait = kind
					trait = humble
					trait = content
					trait = craven
					trait = trusting
				}
			}
			change_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				OR = {
					trait = celibate
					trait = monk
					trait = nun
					trait = chaste
					trait = slothful
					trait = proud
					trait = gluttonous
					trait = is_fat
					trait = is_malnourished
					trait = feeble
					trait = dull
					trait = uncouth
				}
			}
			subtract_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		
		if = {
			limit = {
				OR = {
					trait = strong
					trait = robust
					trait = sturdy
				}
			}
			multiply_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				OR = {
					trait = fair
					trait = groomed
				}
			}
			multiply_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				OR = {
					trait = quick
					trait = genius
					trait = shrewd
				}
			}
			multiply_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				trait = immortal
			}
			change_variable = { which = lf_slave_sell_price value = 50 }
		}
		if = {
			limit = {
				OR = {
					trait = disfigured
					trait = one_eyed
					trait = one_handed
					trait = one_legged
				}
			}
			divide_variable = { which = lf_slave_sell_price value = 2 }
		}
		if = {
			limit = {
				OR = {
					trait = clubfooted
					trait = dwarf
					trait = harelip
					trait = hunchback
					trait = lisp
					trait = stutter
					trait = ugly
					trait = weak
					trait = giant
				}
			}
			divide_variable = { which = lf_slave_sell_price value = 2.5 }
		}
		if = {
			limit = {
				OR = {
					trait = imbecile
					trait = inbred
					trait = slow
					trait = eunuch
				}
			}
			divide_variable = { which = lf_slave_sell_price value = 4 }
		}
		
		# Now we check to see if slave price is lower than the minimum. If so, we set the price to 5.
		if = {
			limit = {
				check_variable = { which = lf_slave_sell_price value < 5 }
			}
			set_variable = { which = lf_slave_sell_price value = 5 }
		}
	}
}
